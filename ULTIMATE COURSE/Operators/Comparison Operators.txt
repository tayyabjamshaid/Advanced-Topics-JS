let x = 6;
let y = "6";
console.log("x == y", x == y); //it will only check the value     //true
console.log("x != y", x != y); //it will only check the value     //false
console.log("x === y", x === y); //it will check the value with datatype   //false
console.log("x !== y", x !== y); //not equal value or  not equal type   //true becoz datatype are different
